plot a sine and cos wave on the same graph

clc;
clear all;
close all;
clf;
t=0:0.0005:2*pi;
x=sin(t);
y=cos(t);
plot(t,x,'r','LineWidth',2);
hold on;
plot(t,y,'b','LineWidth',3);
grid on;
xlabel('time');
ylabel('Amplitude');
title('sin and cos wave');
legend('x','y');
